CREATE DATABASE db_provedoores
use db_provedoores
CREATE TABLE Proveedor (
    id_proveedor INT PRIMARY KEY,
    nombre NVARCHAR(100),
    direccion NVARCHAR(200),
    localidad NVARCHAR(100),
    provincia NVARCHAR(100)
);

CREATE TABLE Categoria (
    id_categoria INT PRIMARY KEY,
    nombre NVARCHAR(50) 
);

CREATE TABLE Pieza (
    id_pieza INT PRIMARY KEY,
    nombre NVARCHAR(100),
    color NVARCHAR(50),
    precio DECIMAL(10, 2),
    id_categoria INT,
    CONSTRAINT FK_Pieza_Categoria FOREIGN KEY (id_categoria) REFERENCES Categoria(id_categoria)
);

CREATE TABLE Suministro (
    id_proveedor INT,
    id_pieza INT,
    fecha DATE,
    cantidad INT,
    PRIMARY KEY (id_proveedor, id_pieza, fecha),
    FOREIGN KEY (id_proveedor) REFERENCES Proveedor(id_proveedor),
    FOREIGN KEY (id_pieza) REFERENCES Pieza(id_pieza)
);
